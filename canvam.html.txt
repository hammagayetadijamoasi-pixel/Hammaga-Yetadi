    <script>
        const tg = window.Telegram.WebApp;
        tg.expand(); // Ilovani to'liq ekranga yoyish

        function switchTab(tabId) {
            // Hamma bo'limlarni yashirish
            document.querySelectorAll('.section').forEach(el => el.classList.remove('active'));
            document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));

            // Tanlanganini ochish
            document.getElementById(tabId).classList.add('active');
            
            // Iconni aktivlashtirish
            const iconIndex = ['home', 'learn', 'admin'].indexOf(tabId);
            document.querySelectorAll('.nav-item')[iconIndex].classList.add('active');
        }

        function sendData() {
            const email = document.getElementById('emailInput').value;
            if(!email.includes('@') || email.length < 5) {
                tg.showPopup({title: 'Xato', message: "Iltimos, to'g'ri email kiriting!"});
                return;
            }
            // Ma'lumotni Python botga yuborish
            tg.sendData(email);
        }
